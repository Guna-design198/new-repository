select * from egp_system_items_interface;
  SELECT
  E.CHANGE_NOTICE,
  U.USER_ID,
  U.USERNAME,
  A.ACTION_ID,
  A.ACTION_TYPE,
  S.STATUS_CODE,
  S.STATUS_NAME,
  A.LAST_UPDATE_DATE,
  E.DESCRIPTION
FROM
  EGO_CHANGE_ACTIONS_B A,
  EGO_CHANGE_STATUSES_VL S,
  EGO_ENGINEERING_CHANGES_VL E,
  PER_USERS U
WHERE
  A.OBJECT_ID1 IN (
    SELECT
      CHANGE_ID
    FROM
      EGO_ENGINEERING_CHANGES_VL
    WHERE
      CHANGE_TYPE_ID =: CHANGE_TYPE_ID
      AND APPROVAL_DATE BETWEEN: FROM_DATE
      AND: TO_DATE
      AND CANCELLATION_DATE IS NULL
  )
  AND E.ASSIGNEE_ID = U.USER_GUID
  AND ACTION_TYPE IN ('REJECT')
  AND A.STATUS_CODE = S.STATUS_CODE
  AND A.STATUS_CODE IN (300000001289110, 300000001289111, 300000001289112)
  AND A.OBJECT_ID1 = E.CHANGE_ID
ORDER BY
  1 DESC;
