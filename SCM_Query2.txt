	EGO_ENGINEERING_CHANGES_B CHG, 
	EGP_ITEM_REVISIONS_B REV 
WHERE
	CHG.CHANGE_NOTICE in(:CHANGE) and 
	CHG.CHANGE_ID = CHLI.CHANGE_ID and 
	CHLI.PK2_VALUE=REV.REVISION_ID and 
	REV.INVENTORY_ITEM_ID = ITM.INVENTORY_ITEM_ID and
	ITM.INVENTORY_ITEM_ID = ITM_TL.INVENTORY_ITEM_ID AND 
	ITM.ORGANIZATION_ID = ITM_TL.ORGANIZATION_ID AND 
	ITM_TL.LANGUAGE = USERENV('LANG');
